169void Manager::backtracking(Graph& graph, std::vector<int>& path, std::vector<int>& min_path, double& minDistance, double &totalDistance, Vertex* currVertex, int startVertex) {
170 if (path.size() == graph.getVertexSet().size() && currVertex->getID() == startVertex) {
171 if (totalDistance < minDistance) {
172 min_path = path;
173 minDistance = totalDistance;
174 }
175 return;
176 }
177
178 for (auto edge : currVertex->getAdj()) {
179 if (!edge->getDest()->isVisited()) {
180 if (totalDistance + edge->getDistance() >= minDistance) {
181 continue;
182 }
183 path.push_back(edge->getDest()->getID());
184 edge->getDest()->setVisited(true);
185 totalDistance += edge->getDistance();
186 backtracking(graph, path, min_path, minDistance, totalDistance, edge->getDest(), startVertex);
187 path.pop_back();
188 edge->getDest()->setVisited(false);
189 totalDistance -= edge->getDistance();
190 }
191 }

237std::vector<Vertex*> Manager::prim(Graph graph) {
238 std::vector<Vertex*> mst;
239 if (graph.getVertexSet().empty()) {
240 // return empty set
241 return graph.getVertexSet();
242 }
243
244 for (auto vert : graph.getVertexSet()) {
245 // set all vertices to infinite distance
246 vert->setDist(std::numeric_limits<double>::infinity());
247 vert->setVisited(false);
248 vert->setPath(nullptr);
249 }
250
251 // first vert. as starting point
252 auto first = graph.getVertexSet().front();
253 first->setDist(0);
254
255 // Friend class (using practical class version)
256 // We need to decrease keys so this is useful
257 MutablePriorityQueue<Vertex> q;
258 q.insert(first);
259
260 while (!q.empty()) {
261 // pick closest unprocessed node
262 auto v = q.extractMin();
263 v->setVisited(true);
264 mst.push_back(v);
265
266 for (auto &e : v->getAdj()) {
267 auto w = e->getDest();
268 if (!w->isVisited()) {
269 auto oldDist = w->getDist();
270 if(e->getDistance() < oldDist) {
271 w->setDist(e->getDistance());
272 w->setPath(e);
273 if (oldDist == std::numeric_limits<double>::infinity()) {
274 q.insert(w);
275 }
276 else {
277 q.decreaseKey(w);
278 }
279 }
280 }
281 }
282 }
283
284 return mst;

361void Manager::triangular(Graph& graph) {
362 clock_t start = clock();
363 std::vector<Vertex*> mst = prim(graph);
364
365 double totalDistance = 0;
366
367 Vertex* previousVertex = nullptr;
368 Vertex* lastVertex = nullptr;
369 std::vector<int> path;
370 bool isConnected;
371
372 for (auto v : mst) {
373 v->setVisited(false);
374 }
375
376 for (auto vertex : mst) {
377 if (previousVertex != nullptr) {
378 isConnected = false;
379 for (auto e : previousVertex->getAdj()) {
380 if (e->getDest() == vertex) {
381 if (!vertex->isVisited()) {
382 totalDistance += e->getDistance();
383 path.push_back(vertex->getID());
384 vertex->setVisited(true);
385 previousVertex = vertex;
386 }
387 isConnected = true;
388 break;
389 }
390 }
391
392 if (!isConnected) {
393 vertex->setVisited(true);
394 totalDistance += getDistanceCoordHaversine(graph, previousVertex, vertex);
395 path.push_back(vertex->getID());
396 previousVertex = vertex;
397 }
398 }
399 else {
400 previousVertex = vertex;
401 path.push_back(vertex->getID());
402 lastVertex = vertex;
403 }
404 }
405
406 path.push_back(0);
407
408 for (auto edge: previousVertex->getAdj()) {
409 if (edge->getDest() == lastVertex) {
410 totalDistance += edge->getDistance();
411 }
412 }
413
414 if (previousVertex != nullptr) {
415 totalDistance += getDistanceCoordHaversine(graph, graph.findVertex(0), lastVertex);
416 }
417
418
419 std::cout << "Path: ";
420 for (auto v : path) {
421 std::cout << v << " ";
422 }
423 std::cout << std::endl;
424
425 std::cout << "Total distance: " << totalDistance << std::endl;
426 std::cout << "Execution time (milliseconds): " << (clock() - start) << "ms" << std::endl;

550Graph Manager::MST(Graph& graph, int startingVert) {
551 Graph newGraph;
552 std::vector<Vertex*> mst = christofidesPrim(graph, startingVert);
553
554 std::unordered_set<Vertex*> visited;
555
556 for (auto vert : mst) {
557 newGraph.addVertex(new Vertex(vert->getID(), 0.0, 0.0));
558 visited.insert(vert);
559 }
560
561 for (auto vert : graph.getVertexSet()) {
562 vert->setVisited(false);
563 }
564
565 for (auto vert : graph.getVertexSet()) {
566 if (vert->getPath() != nullptr) {
567 auto edge = vert->getPath();
568 auto origin = edge->getOrig();
569 auto dest = edge->getDest();
570
571 if (!origin->isVisited() && !dest->isVisited()) {
572 newGraph.addBidirectionalEdge(origin->getID(), dest->getID(), edge->getDistance());
573 visited.insert(origin);
574 visited.insert(dest);
575 }
576 }
577 }
578
579 return newGraph;

623Graph Manager::findMWM(Graph& graph, std::vector<Vertex*>& oddDegree) {
624 Graph mwm;
625
626 for (auto vert : oddDegree) {
627 mwm.addVertex(new Vertex(vert->getID(), 0.0, 0.0));
628 }
629
630 std::vector<bool> isVisited(oddDegree.size(), false);
631 for (int i = 0; i < oddDegree.size(); i++) {
632 if (!isVisited[i]) {
633 double minDist = std::numeric_limits<double>::max();
634 Vertex* minVert = nullptr;
635
636 for (int j = i + 1; j < oddDegree.size(); j++) {
637 if (!isVisited[j]) {
638 double dist = graph.getDistBetween(oddDegree[i]->getID(), oddDegree[j]->getID());
639 if (dist < minDist) {
640 minDist = dist;
641 minVert = oddDegree[j];
642 }
643 }
644 }
645
646 isVisited[i] = true;
647 if (minVert) {
648 isVisited[std::distance(oddDegree.begin(), std::find(oddDegree.begin(), oddDegree.end(), minVert))] = true;
649 mwm.addEdge(oddDegree[i]->getID(), minVert->getID(), minDist);
650 }
651 }
652 }
653
654 return mwm;
655}

684std::vector<int> Manager::findEulerCircuit(Graph& graph) {
685 std::vector<int> euler;
686
687 auto vertexSet = graph.getVertexSet();
688 if (vertexSet.empty()) {
689 // Handle the case where the vertex set is empty
690 return euler; // Returning an empty vector
691 }
692
693 auto first = vertexSet.front();
694
695 std::stack<Vertex*> st;
696 st.push(first);
697 while (!st.empty()) {
698 auto curVert = st.top();
699 if (curVert->getAdj().empty()) {
700 euler.push_back(curVert->getID());
701 st.pop();
702 } else {
703 auto edges = curVert->getAdj();
704 auto edge = edges.front();
705 if (!edge) {
706 // Handle the case where the edge is null
707 st.pop();
708 continue;
709 }
710 auto dest = edge->getDest();
711 if (!dest) {
712 // Handle the case where the destination vertex is null
713 st.pop();
714 continue;
715 }
716 curVert->removeEdge(dest->getID());
717 st.push(dest);
718 }
719 }
720
721 std::reverse(euler.begin(), euler.end());
722
723 return euler;
724}

744
745 if (!hamilton.empty()) {
746 hamilton.push_back(hamilton[0]);
747 }
748

794double Manager::haversine(double lat1, double lon1, double lat2, double lon2) {
795 double delta_lat = convert_to_radians(lat2 - lat1);
796 double delta_lon = convert_to_radians(lon2 - lon1);
797
798 double converted_lat1 = convert_to_radians(lat1);
799 double converted_lat2 = convert_to_radians(lat2);
800
801 // sin^2(delta_lat/2) + cos(rad_lat1) * cos(rad_lat2) * sin^2(delta_lon/2)
802 double aux = std::pow(std::sin(delta_lat / 2), 2) + std::cos(converted_lat1) * std::cos(converted_lat2) * std::pow(std::sin(delta_lon/ 2), 2);
803 double c = 2.0 * std::atan2(std::sqrt(aux), std::sqrt(1 - aux));
804 const double earthRadius = 6371.0;
805 return earthRadius * c;
806}
