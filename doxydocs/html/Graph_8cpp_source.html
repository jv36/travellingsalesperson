14bool Vertex::removeEdge(int in) {
15 bool removedEdge = false;
16 auto it = adj.begin();
17 while (it != adj.end()) {
18 Edge *edge = *it;
19 Vertex *dest = edge->getDest();
20 if (dest->getID() == in) {
21 it = adj.erase(it);
22 deleteEdge(edge);
23 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
24 }
25 else {
26 it++;
27 }
28 }
29 return removedEdge;
30}

334bool Graph::removeVertex(const int &in) {
335 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
336 if ((*it)->getID() == in) {
337 auto v = *it;
338 v->removeOutgoingEdges();
339 for (auto u : vertexSet) {
340 u->removeEdge(v->getID());
341 }
342 vertexSet.erase(it);
343 delete v;
344 return true;
345 }
346 }
347 return false;
348}

398bool Graph::addBidirectionalEdge(const int &sourc, const int &dest, double w) {
399 auto v1 = findVertex(sourc);
400 auto v2 = findVertex(dest);
401 if (v1 == nullptr || v2 == nullptr)
402 return false;
403 auto e1 = v1->addEdge(v2, w);
404 auto e2 = v2->addEdge(v1, w);
405 e1->setReverse(e2);
406 e2->setReverse(e1);
407 return true;

453double Graph::getDistBetween(int source, int dest) const {
454 Vertex* src = findVertex(source);
455 if (src == nullptr) {
456 return std::numeric_limits<double>::max();
457 }
458
459 for (Edge* edge : src->getAdj()) {
460 if (edge->getDest()->getID() == dest) {
461 return edge->getDistance();
462 }
